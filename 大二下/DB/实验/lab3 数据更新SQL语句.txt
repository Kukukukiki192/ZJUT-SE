 zkx_Cno CHAR(8) PRIMARY KEY,
 zkx_Count INT,
 zkx_Avg INT
 )
INSERT　INTO Zhukx_Statistics（zkx_Cno，zkx_Count，zkx_Avg）

FROM Zhukx_Reports
GROUP BY zkx_Cno
